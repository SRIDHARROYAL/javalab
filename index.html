1(b) Roots of a Quadratic Equation

                import java.util.Scanner;

                public class QuadraticEquation {
                    public static void main(String[] args) {
                        Scanner sc = new Scanner(System.in);
                        System.out.print("Enter a, b, c: ");
                        double a = sc.nextDouble();
                        double b = sc.nextDouble();
                        double c = sc.nextDouble();
                        
                        double D = b * b - 4 * a * c;
                        if (D > 0) {
                            System.out.println("Roots are real and different.");
                            System.out.println("Root 1: " + (-b + Math.sqrt(D)) / (2 * a));
                            System.out.println("Root 2: " + (-b - Math.sqrt(D)) / (2 * a));
                        } else if (D == 0) {
                            System.out.println("Roots are real and equal.");
                            System.out.println("Root: " + -b / (2 * a));
                        } else {
                            System.out.println("Roots are complex and different.");
                        }
                        sc.close();
                    }
                }

2(b) Bubble Sort

                public class BubbleSort {
                    public static void main(String[] args) {
                        int[] arr = {64, 25, 12, 22, 11};
                        int n = arr.length;
                        for (int i = 0; i < n - 1; i++) {
                            for (int j = 0; j < n - i - 1; j++) {
                                if (arr[j] > arr[j + 1]) {
                                    int temp = arr[j];
                                    arr[j] = arr[j + 1];
                                    arr[j + 1] = temp;
                                }
                            }
                        }
                        for (int element : arr) {
                            System.out.print(element + " ");
                        }
                    }
                }

4(c) Abstract Class to Find Areas of Different Shapes

                abstract class Shape {
                    abstract void area();
                }

                class Circle extends Shape {
                    double radius = 5.0;

                    void area() {
                        System.out.println("Area of Circle: " + (3.14 * radius * radius));
                    }
                }

                class Rectangle extends Shape {
                    double length = 4.0;
                    double breadth = 6.0;

                    void area() {
                        System.out.println("Area of Rectangle: " + (length * breadth));
                    }
                }

                public class Main {
                    public static void main(String[] args) {
                        Shape circle = new Circle();
                        circle.area();

                        Shape rectangle = new Rectangle();
                        rectangle.area();
                    }
                }

                        }
                    }
                }

                class ThirdThread extends Thread {

                        }

                }

                public class Main {
                    public static void main(String[] args) {
                        new FirstThread().start();
                        new SecondThread().start();
                        new ThirdThread().start();

                }

7(b) isAlive() and join()

                class ThreadExample extends Thread {
                    public void run() {
                        for (int i = 1; i <= 3; i++) {
                            try {
                                Thread.sleep(500);
                            } catch (InterruptedException e) {
                                System.out.println(e);
                            }
                            System.out.println(i);
                        }
                    }

                    public static void main(String args[]) {
                        ThreadExample t1 = new ThreadExample();
                        ThreadExample t2 = new ThreadExample();
                        t1.start();
                        try {
                            t1.join();
                        } catch (Exception e) {
                            System.out.println(e);
                        }
                        t2.start();
                    }
                }

Exercise 9: JDBC

9(a) Connecting to Database

                import java.sql.Connection;
                import java.sql.DriverManager;
                import java.sql.SQLException;

                public class DatabaseConnect {
                    public static void main(String[] args) {
                        String url = "jdbc:mysql://localhost:3306/your_db";
                        String user = "your_username";
                        String password = "your_password";

                        try (Connection conn = DriverManager.getConnection(url, user, password)) {
                            if (conn != null) {
                                System.out.println("Connected to the database!");
                            }
                        } catch (SQLException e) {
                            System.out.println("Connection failed: " + e.getMessage());
                        }
                    }
                }

9(b) Inserting Data into Database

                import java.sql.Connection;
                import java.sql.DriverManager;
                import java.sql.PreparedStatement;
                import java.sql.SQLException;

                public class InsertData {
                    public static void main(String[] args) {
                        String url = "jdbc:mysql://localhost:3306/your_db";
                        String user = "your_username";
                        String password = "your_password";
                        
                        String insertSQL = "INSERT INTO your_table (column1, column2) VALUES (?, ?)";

                        try (Connection conn = DriverManager.getConnection(url, user, password);
                             PreparedStatement pstmt = conn.prepareStatement(insertSQL)) {

                            pstmt.setString(1, "value1");
                            pstmt.setString(2, "value2");
                            pstmt.executeUpdate();

                            System.out.println("Record inserted successfully!");
                        } catch (SQLException e) {
                            System.out.println("Error inserting record: " + e.getMessage());
                        }
                    }
                }
